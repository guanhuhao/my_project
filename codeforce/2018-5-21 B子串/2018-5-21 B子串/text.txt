#include <bits/stdc++.h>
using namespace std;
bool cmp(vector<int> &a, vector<int> &b) {
	return a.size() > b.size();
}
int main()
{
	int n, m;
	vector<int> my[2010];
	scanf("%d%d", &n, &m);
	getchar();
	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < m; j++) {
		char tem;
		scanf("%c", &tem);
		if (tem == '1') my[i].push_back(j);
		}
		getchar();
	}	

	sort(my, my + n, cmp);	
	/*for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < my[i].size(); j++)
			printf("%d  ", my[i][j]);
		cout << endl;
	}*/
	int flag = 0;
	for (int i = 0; i < n; i++) {
		for (int j = n - 1; j > i; j--) {
			for (int k = 0; k < (int)my[j].size(); k++)
			{
				if (find(my[i].begin(), my[i].end(), my[j][k]) == my[i].end()) break;
				else if (k == my[j].size() - 1) {
					/*for (int tem = 0; tem < my[i].size(); tem++) printf("%d  ", my[i][tem]);
					printf("\n");
					for (int tem = 0; tem < my[j].size(); tem++) printf("%d  ", my[j][tem]);
					printf("\n");*/
					flag = 1;
					j = -1;
					i = n + 1;
					break;
				}
			}
		}
	}
	if (flag == 1) printf("YES\n");
	else printf("NO\n");
    return 0;
}
